# gallery/models.py
from django.db import models
import uuid

class GalleryImage(models.Model):
    CATEGORY = [
        ('shirt',  'Shirt / Hoodie'),
        ('poster', 'Poster / Druck'),
        ('card',   'Grußkarte'),
        ('art',    'Fine Art'),
    ]
    CTA = [
        ('etsy',    'Etsy-Listing'),
        ('shop',    'Eigener Shop'),
        ('contact', 'Bestellformular'),
    ]
    id            = models.UUIDField(
                      primary_key=True, default=uuid.uuid4)
    title         = models.CharField(max_length=120)
    slug          = models.SlugField(unique=True)
    description   = models.TextField(blank=True)
    file_path     = models.ImageField(
                      upload_to='gallery/full/')
    thumb_path    = models.ImageField(
                      upload_to='gallery/thumbs/', blank=True)
    category      = models.CharField(
                      max_length=20, choices=CATEGORY)
    tags          = models.CharField(
                      max_length=250, blank=True,
                      help_text='kommasepariert')
    cta_type      = models.CharField(
                      max_length=20, choices=CTA, default='shop')
    cta_url       = models.URLField(
                      blank=True,
                      help_text='Etsy-URL oder Shop-Produktpfad')
    is_public     = models.BooleanField(default=False)
    sort_order    = models.PositiveIntegerField(default=0)
    source_job_id = models.UUIDField(
                      null=True, blank=True,
                      help_text='Referenz auf AIJob')
    created_at    = models.DateTimeField(auto_now_add=True)
    updated_at    = models.DateTimeField(auto_now=True)

    class Meta:
        ordering = ['sort_order', '-created_at']

    def save(self, *args, **kwargs):
        if self.file_path and not self.thumb_path:
            self._generate_thumbnail()
        super().save(*args, **kwargs)

    def _generate_thumbnail(self):
        from PIL import Image, ImageOps
        from io import BytesIO
        from django.core.files.base import ContentFile
        img = Image.open(self.file_path)
        img = ImageOps.exif_transpose(img)
        img.thumbnail((600, 600), Image.LANCZOS)
        buf = BytesIO()
        img.save(buf, format='JPEG', quality=85, optimize=True)
        fname = 'thumb_' + self.file_path.name.split('/')[-1]
        self.thumb_path.save(
            fname, ContentFile(buf.getvalue()), save=False)

Stripe Checkout Session

# shop/views.py
import stripe
from django.conf import settings
from django.http import JsonResponse, HttpResponse
from django.views.decorators.csrf import csrf_exempt

stripe.api_key = settings.STRIPE_SECRET_KEY

def create_checkout_session(request):
    """POST – erstellt Stripe Checkout Session."""
    cart = request.session.get('cart', {})
    line_items = []
    for variant_id, qty in cart.items():
        v = ProductVariant.objects.get(id=variant_id)
        line_items.append({
            'price_data': {
                'currency': 'eur',
                'product_data': {
                    'name': (f"{v.product.name} "
                             f"– {v.size} {v.color}")
                },
                'unit_amount': int(v.price_eur * 100),
            },
            'quantity': qty,
        })
    session = stripe.checkout.Session.create(
        payment_method_types=['card'],
        line_items=line_items,
        mode='payment',
        shipping_address_collection={
            'allowed_countries': ['DE','AT','CH','NL',
                                  'FR','BE']},
        success_url=(settings.BASE_URL
                     + '/shop/success/'
                     '?session_id={CHECKOUT_SESSION_ID}'),
        cancel_url=settings.BASE_URL + '/shop/cart/',
    )
    return JsonResponse({'url': session.url})

Stripe Webhook → Printful

# shop/views.py (Webhook-Handler)
@csrf_exempt   # Stripe sendet keine Django-Session-Cookies
def stripe_webhook(request):
    payload = request.body
    sig     = request.headers.get('Stripe-Signature', '')
    try:
        event = stripe.Webhook.construct_event(
            payload, sig,
            settings.STRIPE_WEBHOOK_SECRET  # aus .env!
        )
    except stripe.error.SignatureVerificationError:
        return HttpResponse(status=400)  # Ablehnen

    if event['type'] == 'checkout.session.completed':
        sess  = event['data']['object']
        order = Order.objects.get(
                    stripe_session_id=sess['id'])
        order.status            = 'paid'
        order.stripe_payment_id = sess['payment_intent']
        addr = sess['shipping_details']['address']
        order.shipping_address  = addr
        order.name = sess['shipping_details']['name']
        order.save()
        create_printful_order.delay(str(order.id))
    return HttpResponse(status=200)

Printful-Order-Task (Celery)

# shop/tasks.py
import requests
from celery import shared_task
from django.conf import settings

@shared_task(bind=True, max_retries=3, queue='cpu_queue')
def create_printful_order(self, order_id):
    order = Order.objects.get(id=order_id)
    items = [
        {'sync_variant_id': line.variant.printful_variant_id,
         'quantity': line.quantity}
        for line in order.lines.all()
    ]
    payload = {
        'recipient': {
            'name':         order.name,
            'email':        order.email,
            'address1':     order.shipping_address['line1'],
            'city':         order.shipping_address['city'],
            'zip':          order.shipping_address[
                                'postal_code'],
            'country_code': order.shipping_address[
                                'country'],
        },
        'items': items,
    }
    try:
        r = requests.post(
            'https://api.printful.com/orders',
            json=payload,
            headers={'Authorization':
                     f"Bearer {settings.PRINTFUL_API_KEY}"},
            timeout=15,
        )
        r.raise_for_status()
        pf = r.json()['result']
        order.printful_order_id = str(pf['id'])
        order.status = 'processing'
        order.save()
    except Exception as exc:
        raise self.retry(exc=exc, countdown=60)

Etsy Open API v3 — Listing-Helper

# etsy/listing.py
import requests
from django.conf import settings

ETSY_BASE = "https://openapi.etsy.com/v3/application"

def create_listing(access_token: str,
                   shop_id: str,
                   product: dict) -> dict:
    """Erstellt ein Etsy-Draft-Listing."""
    headers = {
        "x-api-key":     settings.ETSY_API_KEY,
        "Authorization": f"Bearer {access_token}",
        "Content-Type":  "application/json",
    }
    payload = {
        "title":              product["title"],
        "description":        product["description"],
        "price":              product["price_eur"],
        "quantity":           999,   # on-demand
        "who_made":           "i_did",
        "when_made":          "made_to_order",
        "taxonomy_id":        68887659,  # Clothing > Shirts
        "shipping_profile_id":product["shipping_profile_id"],
        "state":              "draft",
        "tags":               product["tags"][:13],
    }
    r = requests.post(
        f"{ETSY_BASE}/shops/{shop_id}/listings",
        json=payload, headers=headers, timeout=10)
    r.raise_for_status()
    return r.json()

def upload_listing_image(access_token, shop_id,
                         listing_id, image_path):
    """Lädt das Produktbild für ein Etsy-Listing hoch."""
    headers = {
        "x-api-key":     settings.ETSY_API_KEY,
        "Authorization": f"Bearer {access_token}",
    }
    with open(image_path, 'rb') as f:
        r = requests.post(
            f"{ETSY_BASE}/shops/{shop_id}/"
            f"listings/{listing_id}/images",
            headers=headers,
            files={"image": f},
            timeout=30)
    r.raise_for_status()

def activate_listing(access_token, shop_id, listing_id):
    """Setzt Draft-Listing auf 'active'."""
    headers = {
        "x-api-key":     settings.ETSY_API_KEY,
        "Authorization": f"Bearer {access_token}",
        "Content-Type":  "application/json",
    }
    r = requests.patch(
        f"{ETSY_BASE}/shops/{shop_id}/"
        f"listings/{listing_id}",

# gpu/export.py
from PIL import Image, ImageCms
import subprocess, os

def export_pod(src: str, job_id: str):
    """PNG 300dpi sRGB — Printful / POD-Partner."""
    img = Image.open(src).convert('RGB')
    out = f"/mnt/agency_nas/exports/pod/{job_id}.png"
    img.save(out, format='PNG', dpi=(300, 300),
             optimize=True)

def export_offset(src: str, job_id: str):
    """CMYK TIFF → PDF/X-4 mit 3mm Bleed
    für Offset-Druck (FLYERALARM, Saxoprint)."""
    srgb = ImageCms.createProfile('sRGB')
    cmyk = ImageCms.getOpenProfile(
        '/opt/icc/ISOcoated_v2_eci.icc')  # FOGRA39
    img_rgb  = Image.open(src).convert('RGB')
    img_cmyk = ImageCms.profileToProfile(
        img_rgb, srgb, cmyk, outputMode='CMYK')
    tif = f"/mnt/agency_nas/exports/offset/{job_id}.tif"
    img_cmyk.save(tif, format='TIFF', dpi=(300, 300))
    # TIF → PDF/X-4 via Ghostscript
    pdf = tif.replace('.tif', '.pdf')
    subprocess.run([
        'gs', '-dBATCH', '-dNOPAUSE',
        '-sDEVICE=pdfwrite',
        '-dPDFX', '-dPDFSETTINGS=/prepress',
        f'-sOutputFile={pdf}', tif
    ], check=True, timeout=120)

def export_preview(src: str, job_id: str):
    """JPG 72dpi max 1200px — Web / Galerie."""
    img = Image.open(src).convert('RGB')
    img.thumbnail((1200, 1200), Image.LANCZOS)
    out = (f"/mnt/agency_nas/exports/preview/"
           f"{job_id}_preview.jpg")
    img.save(out, format='JPEG', quality=85,
             optimize=True, dpi=(72, 72))

Architektur-Entscheidungen (ADR)

Jede Entscheidung ist dokumentiert: Kontext, Begründung und Konsequenzen. Accepted = bindend   Proposed = noch offen   Deprecated = ersetzt. Neue Subsysteme brauchen eine neue ADR.

ADR-01 Django 4.2 LTS als Web-Framework (nicht FastAPI / Node.js) Accepted

Benötigt: REST-API, Admin-Interface, Auth, File-Handling, ORM in einer Einheit. Team-Skill in Python vorhanden. ML-Ökosystem (Celery, PIL, ICC) ist Python-first. 2-Personen-Team braucht minimalen Boilerplate.

Django 4.2 LTS (Security-Support bis April 2026). LTS-Upgrade auf Django 5.2 LTS geplant für Q2 2025. Kein Django REST Framework nötig für MVP – Django-Views reichen für Webhooks und einfache APIs.

Admin out-of-the-box (kein separates Tool). ORM mit Migrations reduziert Fehler. Bestehende Django-Projekte auf VPS wiederverwendbar. Django Security-Defaults sind stark (CSRF, XSS, SQLi).

Mehr Overhead als FastAPI für reine API-Endpoints. Monolith erfordert klare App-Trennung (gallery/, shop/, gpu/) um Skalierbarkeit offen zu halten. Kein Auto-Scaling im MVP.

ADR-02 PostgreSQL 15 als primäre Datenbank Accepted

Metadaten für AI-Jobs, Bestellungen und Galerie-Bilder sind relational strukturiert. Teile (Prompt-Parameter, Exportpfade, Shipping-Adresse) sind semistrukturiert. DSGVO erfordert EU-Datenhaltung und Löschkonzept.

PostgreSQL 15 auf dem Hetzner VPS (DE-Datacenter). Bietet JSONField für semistrukturierte Daten, ACID-Transaktionen für Bezahl-Flow, UUID-Primary-Keys nativ, DSGVO-konformes Soft-Delete via is_deleted-Flag.

JSONField ersetzt separate Metadata-Tabellen. UUID verhindert sequenzielle ID-Enumeration (Sicherheit). pg_dump für tägliche Backups trivial einzurichten. Bewährt in Django-Ökosystem.

Single-Node PostgreSQL – kein HA. Für MVP ausreichend; Read-Replica erst nötig ab signifikantem Traffic. pg_hba.conf muss auf localhost beschränkt bleiben.

ADR-03 Celery 5 + Redis 7 als Task-Queue (zwei Queues) Accepted

GPU-Rendering, Bild-Export und Printful-Bestellungen dürfen den HTTP-Request-Cycle nicht blockieren. Zwei verschiedene Worker-Typen nötig: GPU-Tasks (1 concurrent, RunPod-bound) und CPU-Tasks (2 concurrent, Export+API).

Celery 5.x mit Redis 7 als Broker + Result Backend. Zwei Queues: gpu_queue (1 Worker) und cpu_queue (2 Worker). Retry-Decorator mit exponential backoff. Redis appendonly yes für Queue-Persistenz.

Entkopplung GPU-Calls vom Web-Prozess. Retry-Logik verhindert verlorene Jobs. Flower-Dashboard für Queue-Monitoring. Skalierung: weitere Worker ohne Architekturänderung.

Redis ohne appendonly yes verliert Queue bei Neustart. Redis ist Single-Node – Ausfall = Queue-Ausfall. Lösung: Redis Sentinel bei Wachstum. Celery-Worker-Logs müssen überwacht werden.

ADR-04 RunPod EU (primär) + Vast.ai (Fallback) statt eigenem GPU-Server Accepted

Kein eigener GPU vorhanden. RTX 4090 Kaufpreis ~€1.800. Bei sporadischem Rendering-Bedarf im MVP lohnt CapEx nicht. GPU-Rendering ist async (Queue) – Latenz unkritisch. EU-Datacenter für DSGVO-Nähe bevorzugt.

RunPod EU (Frankfurt/Nürnberg) primär: RTX 4090 ~€0.40/h on-demand, stündliche Abrechnung, keine Mindestlaufzeit. Vast.ai als automatischer Fallback im Celery-Task wenn RunPod unavailable.

Kein Hardware-CapEx, kein Betrieb/Kühlung/Strom. Kosten direkt proportional zu Nutzung (~€0.01–0.02/Bild). Kein Lock-in – bei Wachstum eigenen GPU-Server nachziehen (ADR-04b).

Abhängigkeit von zwei Drittanbietern. Netzwerk-Latenz für große Outputs (~100–500MB). Lösung: Outputs direkt auf RunPod-Storage zwischenspeichern, nur URL zurückgeben, dann selektiv herunterladen.

ADR-05 Printful als normiertes Fulfillment-Backend (alle Kanäle) Accepted

Mehrere Vertriebskanäle (Etsy, Shop, B2B). Ohne normierten Fulfillment-Partner müssten Druckdateien kanalspezifisch aufbereitet werden. 2-Personen-Team hat keine Kapazität für eigenes Lager/Versand. Printify als Alternative mit ähnlicher API.

Printful als einziges Fulfillment-Backend: eine Druckdatei → Printful übernimmt Druck, Verpackung, Versand für alle Kanäle. Printify als Backup konfiguriert (ähnliche API-Struktur, einfache Migration möglich).

Normierte Druckdatei einmal erstellen, überall verwenden. DE-Lager → Lieferung DE/AT/CH 3–5 Tage. Kein Lagerrisiko. Printful↔Etsy-Integration fertig out-of-the-box. Qualitätskonstanz über alle Kanäle.

Printful-Produktkosten reduzieren Marge vs. eigenem Druck. Qualität abhängig vom Printful-Drucknetzwerk. Migration zu Printify möglich wenn Preise unvorteilhaft werden.

ADR-06 Stripe Checkout als Zahlungsanbieter (Hosted Sessions + Webhooks) Accepted

EU-Zahlungen erfordern PSD2/SCA (Strong Customer Authentication). Eigene Kartendaten-Verarbeitung würde PCI DSS Level 1 erfordern – unrealistisch für 2-Personen-Team. Zuverlässige Webhooks kritisch für Printful-Trigger.

Stripe Checkout (Hosted): Kartendaten landen nie auf eigenem Server → PCI SAQ A (einfachste Stufe). Webhook-Signatur-Validierung mit STRIPE_WEBHOOK_SECRET. Vollständige Testmodus-Unterstützung für E2E-Tests.

PCI-Compliance out-of-the-box. SCA/3DS automatisch. Zuverlässige Webhooks (Stripe Retry-Logik). EUR-native. Stripe-Dashboard für Transaktions-Übersicht. Keine eigene Zahlungsseite nötig.

2.9% + €0.25/Transaktion. Stripe-Outage beeinflusst Checkout (Monitoring: status.stripe.com). Kein eigenes Checkout-UI möglich (bewusst akzeptiert für PCI-Compliance).

ADR-07 Synology DS925+ als primäres File-Storage via NFS v4 + Tailscale Accepted

GPU-Outputs sind groß (100–500MB/Bild nach Upscale). S3-Storage wäre bei Volumen teuer. NAS ist bereits vorhanden, kein zusätzlicher CapEx. Verbindung VPS↔Home-NAS muss sicher und performant sein.

Synology DS925+ via NFS v4 auf VPS gemountet (/mnt/agency_nas/). Verbindung über Tailscale VPN (Mesh-VPN, kein offenes NFS im Internet). Synology Hyper Backup für Offsite. RAID für lokale Redundanz.

CapEx bereits amortisiert. Große Files ohne Cloud-Storage-Kosten. Täglich Snapshots. Django sieht NAS als normales lokales Dateisystem – kein SDK nötig. Tailscale verschlüsselt die NFS-Verbindung.

NAS + Heimanschluss sind Single Points of Failure für File-Storage. Empfehlung: kritische Exports (final_pod/) zusätzlich auf Hetzner-Volume spiegeln. Tailscale muss auf beiden Nodes laufen.

ADR-08 Hetzner VPS (Nürnberg / Falkenstein) als App-Server Accepted

Django, PostgreSQL, Redis, Celery und Nginx auf einem Server. DSGVO erfordert EU-Datenhaltung für Nutzerdaten (Bestellungen, E-Mails). Bestehender Hetzner VPS aus anderen Django-Projekten verfügbar.

Hetzner Cloud VPS (CX21 oder CX31), Rechenzentrum Nürnberg/Falkenstein (Deutschland). Ubuntu 22.04 LTS. Hetzner ist ISO 27001-zertifiziert, DSGVO-konform, dedizierte IPv4+IPv6. ~€10–20/Monat.

DSGVO: Nutzerdaten in Deutschland. Günstiger als AWS/GCP. Stack bereits bekannt (bestehende Projekte). Hetzner Managed Backups optional (+€2/Mo). Snapshot für schnelles Rollback.

Kein Auto-Scaling – Spikes erfordern manuelles Resize. Kein managed Kubernetes – alles manuell. Monitoring (Hetzner Metrics + UptimeRobot) zwingend empfohlen um Ausfälle frühzeitig zu erkennen.

Jeder Job basiert auf einem PipelineTemplate, das die auszuführenden Schritte als Boolean-Flags speichert. Studio-Mitarbeiter wählen das Template im Job-Formular; Admin kann Templates im Django-Admin verwalten.

Job-Lifecycle

  1. Studio-Mitarbeiter erstellt Job → Status: draft
  2. Admin prüft und startet via Custom Action → Status: queued
  3. Celery-Chain startet automatisch → Status: running
  4. Pipeline-Steps laufen durch → JobStep.status zeigt Fortschritt
  5. HTMX-Polling zeigt Live-Status im Browser (alle 3s, kein Page-Reload)
  6. Pipeline fertig → Status: done, Studio-Mitarbeiter sieht Ergebnisse
  7. Asset-Selektion: Mitarbeiter markiert beste Bilder für Galerie/Shop/Etsy

Partner & Channels — Multi-Shop & Druckpartner-System

Das Grundprinzip: Bestellungen kommen über verschiedene Vertriebskanäle rein — eigener Shop, Etsy, externe WooCommerce/Shopify-Shops oder manuelle Aufträge. Jede Bestellung wird automatisch an den zuständigen Druckpartner weitergeleitet, der die passenden Druckdateien bekommt und direkt an den Endkunden verschickt. Das ist das Standard Drop-Shipping Modell — wie Printful, nur für lokale Druckstudios.

Kern-Flow

Bestellung eingehend (beliebiger Kanal)
    ↓
  Order-Objekt in DB  (channel FK auf SalesChannel)
    ↓
  Fulfillment-Partner zuweisen (via Admin oder Produkt-Routing-Regel)
    ↓
  create_print_bundle Task  →  ZIP mit Druckdateien auf NAS
    ↓
  notify_fulfillment_partner Task  →  E-Mail + optionaler Webhook
    ↓
  Partner loggt ein, sieht Auftrag, lädt Bundle herunter
    ↓
  Partner druckt und verschickt  →  Tracking-Code einpflegen  →  Status: shipped

SalesChannel — Vertriebskanäle

Jeder Kanal (eigener Shop, Etsy, externer WooCommerce-Shop, manuell) wird als SalesChannel-Objekt in der DB gespeichert. Order.channel FK verbindet Bestellung mit Kanal.

channel_typeBeispielIntegration
own_shopdatemyhobby.com/shopStripe Webhook (bereits gebaut)
etsyEtsy StoreEtsy Open API v3 (etsy/ App)
woocommerceWordPress-Partner-ShopWooCommerce REST API + webhook_url
shopifyShopify-Partner-ShopShopify Webhooks + api_key/secret
manualTelefon/E-Mail-BestellungManuell via Admin eingepflegt

FulfillmentPartner — Druckstudios

Jedes Druckstudio ist ein FulfillmentPartner mit eigenem Django-Login (Gruppe: print_partners). Admin weist Produkte via PartnerProduct zu. Pro Produkt wird festgelegt, welche Dateitypen der Partner erhält.

FeldInhaltBeispiel

PartnerVariant — Welche Varianten kann der Partner produzieren?

Granularere Kontrolle: nicht nur "Partner kann dieses Produkt" sondern "Partner kann Poster in A2 und A3 aber nicht A4". Wenn kein PartnerVariant-Eintrag existiert, kann der Partner die Variante nicht produzieren.

PartnerVariant:
  partner       FulfillmentPartner
  variant       shop.ProductVariant   # z.B. "Poster DIN A2", "T-Shirt XL schwarz"
  partner_sku   "DRUCK-A2-001"       # interne Bestellnummer des Studios
  is_available  True/False            # kurzfristig nicht lieferbar?

PartnerProduct — Routing-Tabelle

Welcher Partner bekommt welche Produkte — und mit welchen Dateitypen? Verwaltet im Admin. Grundregel: ein Produkt kann mehreren Partnern zugewiesen sein (z.B. Backup-Partner), aber in der Praxis meist 1:1.

PartnerProduct:
  partner      FulfillmentPartner
  product      shop.Product
  export_types ['cmyk', 'pod']   # subset von ProductVariant.required_export_types

ImageProduct — Bild ↔ Produkt Zuordnung + Mockup

Verbindet ein Galerie-Bild mit einem Produkt. Enthält das vorausberechnete Mockup-Bild (Bild auf Produkt platziert). Wird angelegt wenn ein Studio-Mitarbeiter ein Bild für den Shop freigibt. Löst sofort generate_all_mockups aus.

ImageProduct:
  image          GalleryImage
  product        shop.Product
  mockup_file    "shop/mockups/abc123-poster-a2.jpg"  # Printful Mockup API Output
  mockup_status  pending → generating → ready / failed
  is_primary     True = Haupt-Produktbild im Shop

Kunden-Konfigurator Flow

Kunde sieht Bild in Galerie → klickt "Kaufen als Produkt" → /shop/configure/<slug>/ zeigt alle verknüpften Produkte mit fertigen Mockups → Kunde wählt Größe/Farbe → Warenkorb → Stripe Checkout.

  1. Studio-Mitarbeiter selektiert Bild via AssetSelectView (POST)
  2. System legt ImageProduct-Einträge für alle passenden Produkte an
  3. Celery Task generate_all_mockups ruft Printful Mockup API auf für jede Variante
  4. ImageProduct.mockup_status wechselt zu ready wenn fertig
  5. Galerie-Bild zeigt CTA "Als Produkt bestellen" → Konfigurator-URL
  6. Kunde wählt Produkt + Variante → Mockup wird als Produktbild angezeigt
